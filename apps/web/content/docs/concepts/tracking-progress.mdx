---
title: Tracking Response Progress
description: Show users what Tambo is doing during response generation.
---

You can use a thread's `generationStage` property to display the specific stage of Tambo's response generation process to give users better insight into what's happening.

```tsx
const { thread } = useTambo();

// Display the current generation stage
const getStageMessage = (stage: string) => {
  switch (stage) {
    case "thinking":
      return "Tambo is thinking...";
    case "planning":
      return "Planning the response...";
    case "generating":
      return "Generating response...";
    case "rendering":
      return "Rendering components...";
    default:
      return "Processing...";
  }
};

// In your component
{
  thread.generationStage && (
    <div className="progress-indicator">
      <span>{getStageMessage(thread.generationStage)}</span>
    </div>
  );
}
```
