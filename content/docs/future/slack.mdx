## Getting Started

### 1. Install Dependencies

```bash tab="pnpm"

```
</Tabs>

### 2. Initialize Hydra and Bolt

```typescript
import { HydraClient } from "hydra-ai";
import { HydraBoltManager } from "hydra-ai/bolt";

// Initialize Hydra
const hydraClient = new HydraClient("gpt-4", "openai");

// Initialize Bolt integration
const boltManager = new HydraBoltManager(
  process.env.SLACK_BOT_TOKEN!,
  process.env.SLACK_SIGNING_SECRET!,
  hydraClient
);

// Start the Bolt server
await boltManager.start(3000);
```

### 3. Define Interactive Components

```typescript
import { jsxslack } from "jsx-slack";
import { Modal, Input, Section } from "jsx-slack";

export const feedbackModal = () => jsxslack`
  <Modal 
    title="Feedback Form" 
    callbackId="feedback_submission"
    close="Cancel"
  >
    <Section>Share your thoughts:</Section>
    <Input 
      type="text" 
      blockId="name_block"
      actionId="user_name"
      label="Name" 
      required 
    />
    <Input 
      type="textarea" 
      blockId="feedback_block"
      actionId="feedback"
      label="Your Feedback" 
      required 
    />
    <Input type="submit" value="Submit" />
  </Modal>
`;

// Register with Hydra
await hydraClient.registerComponent(
  "feedbackModal",
  "Interactive feedback collection form",
  feedbackModal
);
```

### 4. Handle Interactive Actions

```typescript
// Register action handler
boltManager.registerActionHandler(
  "feedback_submission",
  async ({ view, body, hydraClient }) => {
    const userName = view.state.values.name_block.user_name.value;
    const feedback = view.state.values.feedback_block.feedback.value;

    // Process with Hydra
    const response = await hydraClient.generateComponent(
      `Create response for ${userName}'s feedback: ${feedback}`
    );

    // Send response
    await boltManager.getBoltApp().client.chat.postMessage({
      channel: body.user.id,
      blocks: response.component,
    });
  }
);
```

### 4. Handle Interactive Actions with API Integration

```typescript
// Register action handler with API integration
boltManager.registerActionHandler(
  "process_request",
  async ({ view, body, hydraClient }) => {
    try {
      // Call your API
      const apiResponse = await fetch("https://api.example.com/process", {
        method: "POST",
        body: JSON.stringify(view.state.values),
      });
      const result = await apiResponse.json();

      // Let Hydra analyze the API response and suggest follow-up actions
      const aiResponse = await hydraClient.generateComponent(
        "Analyze API response and suggest next steps",
        {
          context: {
            apiResult: result,
            userIntent: view.state.values,
            previousActions: body.actions,
          }
        }
      );

      // Send the AI-generated response with suggested actions
      await boltManager.getBoltApp().client.chat.postMessage({
        channel: body.user.id,
        blocks: [
          ...aiResponse.component, // AI-generated response with context
          {
            type: "actions",
            elements: [
              {
                type: "button",
                text: {
                  type: "plain_text",
                  text: "View Details",
                },
                value: "view_details",
                action_id: "view_api_details"
              },
              {
                type: "button",
                text: {
                  type: "plain_text",
                  text: "Take Suggested Action",
                },
                value: JSON.stringify(aiResponse.suggestedAction),
                action_id: "execute_suggestion"
              }
            ]
          }
        ]
      });
    } catch (error) {
      // Handle error appropriately
      console.error("API or AI processing failed:", error);
    }
  }
);
```

### 5. Chain Multiple Actions

```typescript
// Handle the suggested action execution
boltManager.registerActionHandler(
  "execute_suggestion",
  async ({ action, hydraClient }) => {
    const suggestion = JSON.parse(action.value);
    
    // Execute the suggested action
    const result = await hydraClient.executeAction(suggestion);
    
    // Generate a new response based on the action result
    const nextResponse = await hydraClient.generateComponent(
      "Generate response for action result",
      {
        context: {
          actionResult: result,
          previousSuggestion: suggestion
        }
      }
    );

    // Send the follow-up response
    await boltManager.getBoltApp().client.chat.postMessage({
      channel: action.channel.id,
      blocks: nextResponse.component
    });
  }
);
```

## Advanced Features

### Context Tools Integration

```typescript
await hydraClient.registerComponent(
  "userList",
  "Dynamic user list display",
  (props) => jsxslack`
    <Blocks>
      <Section>
        <b>Active Users:</b>
        ${props.users.map((user) => `<Section>${user}</Section>`).join("")}
      </Section>
    </Blocks>
  `,
  { users: { type: "array", required: true } },
  [
    {
      definition: {
        name: "fetchUsers",
        description: "Get active users",
        parameters: [],
      },
      getComponentContext: async () => {
        const response = await fetch("https://api.example.com/users");
        return response.json();
      },
    },
  ]
);
```

### Error Handling

<Callout type="info" title="Best Practice">
  Always implement comprehensive error handling to ensure a smooth user
  experience.
</Callout>

```typescript
boltManager.registerActionHandler(
  "critical_action",
  async ({ action, hydraClient }) => {
    try {
      const result = await hydraClient.generateComponent(action.value);
      // Handle success
    } catch (error) {
      // Handle error gracefully
      console.error("Action processing failed:", error);
      await boltManager.getBoltApp().client.chat.postMessage({
        channel: action.channel.id,
        text: "Unable to process action. Please try again.",
      });
    }
  }
);
```
